procedure minimum(L: liste)
	mini: entier
	n: entier ; n = longueur(L)
	i: entier
	si n = 0 alors:
		return 0
	mini = L[1]
	pour i allant de 2 à n faire:
		si L[i] < mini alors:
			mini = L[i]
	retourner mini
